    ```python
    def test_malformed_yaml_raises_permanent_error():
        """OP-11: malformed YAML in ConfigMap valuesFiles -> kopf.PermanentError chained from yaml.YAMLError.

        Mirrors Test 9's CR shape (helmChart + valuesFiles -> ConfigMap), but:
          - The CM mock SUCCEEDS (no NotFoundError) and returns data with malformed YAML.
          - The render path is exercised by passing variables=None-equivalent
            (no `variables:` block) so render() short-circuits via pre-scan guard
            and yaml.safe_load is the failure point.
          - Acceptance: kopf.PermanentError is raised; the underlying cause chain
            includes yaml.YAMLError.
        """
        import kopf
        import yaml
        from main import handle_appstack_deployment

        # Malformed YAML: unclosed flow-style mapping after a valid key. Picked
        # because yaml.safe_load(values_yaml) raises yaml.YAMLError (subclass
        # ScannerError or ParserError) reliably across PyYAML versions.
        malformed_cm = _make_kr8s_cm({
            "values.yaml": "key: value\nbad: {unclosed: [list, of, items\n",
        })

        components = [{
            "name": "vector-db",
            "helmChart": {
                "repository": "oci://example.registry/charts",
                "name": "milvus",
                "version": "4.2.1",
            },
            "valuesFiles": [
                {"kind": "ConfigMap", "name": "malformed-cm", "key": "values.yaml"},
            ],
        }]
        # No `variables:` block -> render() short-circuits (pre-scan guard sees no '${')
        # and yaml.safe_load is the exclusive failure point.
        spec = {"appStack": {"components": components}}

        with patch("main.subprocess.run"), \
             patch("main._load_kube_config_once", return_value=False), \
             patch("main.kr8s.objects.ConfigMap.get", return_value=malformed_cm), \
             patch("main.HelmOperator"), \
             pytest.raises(kopf.PermanentError) as exc_info:
            handle_appstack_deployment(
                body={"spec": spec},
                spec=spec,
                name="ai-research",
                namespace="staging",
                status={},
            )

        # Plan 01 Task 2 chains yaml.YAMLError -> kopf.PermanentError via `raise ... from e`.
        cause = exc_info.value.__cause__
        assert isinstance(cause, yaml.YAMLError), (
            f"Expected kopf.PermanentError chained from yaml.YAMLError; "
            f"got chain ending with {type(cause).__name__}: {cause!r}"
        )
        assert "malformed-cm" in str(exc_info.value)
    ```
