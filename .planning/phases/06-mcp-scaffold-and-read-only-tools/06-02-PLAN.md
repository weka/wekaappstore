```yaml
    components:
      - name: component-name
        enabled: true
        depends_on: []
        target_namespace: ns
        helm_chart:
          repository: https://...
          name: chart-name
          version: "1.0.0"
          release_name: release
          crds_strategy: Auto
        values: {}
        wait_for_ready: true
        readiness_check:
          type: deployment
          name: deploy-name
          timeout: 600
  prerequisites: []
```
